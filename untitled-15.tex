\documentclass[tikz]{standalone}
\usepackage{luacode}
\begin{document}
\begin{luacode}
local function G (x) 
    return string.format('%  .2f',-(x^2)/3)
end

tex.print('\\begin{tikzpicture}[scale=.3]')
tex.print('\\draw[color=red] (0,0)')

local z=-.1
for i=1,50 do
   tex.print('-- (' .. i .. ',' .. G(z)..')  ') 
   z=z+.2
end

tex.print(';')
tex.print('\\end{tikzpicture}')
\end{luacode}
\end{document}
